class A {
	public int v;
	
	public void setV(int _v) {
		this.v = _v;
	}
	
	public int getV() {
		return this.v;
	}
	
	public A() {
	    this.v = 1;
	}
	
}

public class Main {

	public static void main(String args[]) {
		A a1 = new A();
		print a1.v;
		print a1.getV();
		a1.v = 2;
		print a1.v;
		a1.setV( 3 );
		print a1.getV();
		print a1.getV();
		a1.setV( 5 );
	}
}
